Examples
--------

.. code-block:: yaml

    # fetches all existing printers
    - name: Pulling existing printers
      nils_ost.bambuddy.list:
        url: "{{ bambuddy.url }}"
        token: "{{ bambuddy.token }}"
        target: printer
      delegate_to: localhost
      register: existing_printers
